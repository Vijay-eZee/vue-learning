/**
 * Person class for person detail
 */
class Person {
    /**
     * 
     * @type {string} 
     * @type {number} 
     */
    /**
     * 
     * @param {string} name -person name
     * @param {number} age -person age
     */
    constructor(name,age) {
        this.name = name
        this.age=age
    }
    /**
     * @property {Function} displayDetail display detail
     * @returns void
     */
    displayDetail() {
        console.log(`name is ${this.name} and age is ${this.age}`)
    }
}
